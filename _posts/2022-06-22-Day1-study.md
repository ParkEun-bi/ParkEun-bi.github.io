# 프로그래밍 문제 - java

* 2017년도 1회차

```java
public class Test1{
  public static void main(String[] args){
    int[] a = {3, 4, 10, 2, 5};
    int temp;
    
    for(int i=0; i<=3; i++){
      for(int j=i+1; j<=4; j++){
        if(a[i] < a[j]){
          temp = a[i];
          a[i] = a[j];
          a[j] = temp;
        }
      }
    }
    for(int i=0; i<5; i++){
      System.out.println(a[i]);
    }
}
```
